Table Gimnasio {
  ID_Gimnasio int [pk, increment]
  Nombre varchar
  Direccion varchar
  Num_Telefono varchar
}

Table Rol {
  ID_Rol int [pk, increment]
  Nombre_Rol varchar   // AdminGeneral, AdminSucursal, Empleado
  Descripcion varchar
}

Table Empleado {
  ID_Empleado int [pk, increment]
  ID_Gimnasio int       // FK
  Nombre varchar
  APaterno varchar
  AMaterno varchar
  Num_Telefono varchar
  Direccion varchar
  Fecha_Nac date
  Fecha_Ingreso date
}

Table Usuario {
  ID_Usuario int [pk, increment]
  ID_Empleado int       // FK
  ID_Rol int            // FK
  Usuario varchar
  PasswordHash varchar
  Estado varchar        // Activo / Inactivo
  Fecha_Creacion datetime
  Ultimo_Login datetime
}

Table Miembro {
  ID_Miembro int [pk, increment]
  ID_Gimnasio int       // Gimnasio donde se registró
  Nombre varchar
  APaterno varchar
  AMaterno varchar
  Num_Telefono varchar
  Direccion varchar
  Fecha_Nac date
  Fecha_Ingreso date
  Foto varchar
}

Table Membresia {
  ID_Membresia int [pk, increment]
  Modalidad varchar      // Diaria, Semanal, Mensual
  Precio decimal
}

Table Inscripcion {
  ID_Inscripcion int [pk, increment]
  ID_Miembro int        // FK
  ID_Membresia int      // FK
  ID_Gimnasio_Pago int  // FK (dónde se pagó)
  Fecha_Inicio date
  Fecha_Fin date
  Estado varchar        // Activa, Expirada, Cancelada
}


Table Producto {
  ID_Producto int [pk, increment]
  Nombre varchar
  Cantidad int
  Precio decimal
}

Table MovimientoInventario {
  ID_Mov int [pk, increment]
  ID_Producto int       // FK
  ID_Empleado int       // FK (quién movió el inventario)
  ID_Gimnasio int       // FK
  Tipo varchar          // Entrada, Salida, Ajuste
  Cantidad int
  Fecha datetime
  Motivo varchar
}
